
//Initializing git repo 
git init
git add <>
git commit -m "<>"          /// -m means message
git branch -M main          /// -M Master
git remote add origin <>    
git push -u origin main     /// -u upstream

// Adding Updates to the git repo
git add .
git commit -m "description"
git push origin main 

// Miscelleneous

git status -> to check staged and unstaged files, current branch
git add -> adds files to be commited
git commit -> commits file to git
git push -> pushes files to github
